package tinycsp;

/**
 * Implementation of the variable
 */
public class Variable {

    Domain dom;

    /**
     * Creates a variable with some values
     * @param n a positive number such that {0..n-1} is the domain variable
     */
    public Variable(int n) {
        dom = new Domain(n);
    }
}
